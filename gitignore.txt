// Add Save & Clear buttons to filters
  const filterActions = document.createElement("div");
  filterActions.classList.add("filter-actions");

  const saveBtn = document.createElement("button");
  saveBtn.textContent = "Save";
  saveBtn.addEventListener("click", () => {
    const selected = [...categorySection.querySelectorAll("input:checked")]
      .map(cb => cb.parentElement.textContent.trim());
    alert("Saved filters: " + selected.join(", "));
  });

  const clearBtn = document.createElement("button");
  clearBtn.textContent = "Clear";
  clearBtn.addEventListener("click", () => {
    categorySection.querySelectorAll("input").forEach(cb => (cb.checked = false));
  });

  filterActions.appendChild(saveBtn);
  filterActions.appendChild(clearBtn);
  categorySection.appendChild(filterActions);
